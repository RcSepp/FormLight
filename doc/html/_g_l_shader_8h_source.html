261  {

267  case 'f': case 'p': ++nfs; break;
268  case 'g': ++ngs; break;

271  return;
272  }

274  va_list args;

276  nvs = nfs = ngs = 0;
277  for (const char* c = shaderTypes; *c; ++c)
278  switch (*c)
279  {

281  case 'f': case 'p': fs[nfs++] = va_arg(args, const char*); break;
282  case 'g': gs[ngs++] = va_arg(args, const char*); break;
283  }
284  va_end(args);
285

292  if (isCompiled == 0)
